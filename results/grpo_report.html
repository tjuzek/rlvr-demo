def is_undulating(num):
    # Function to check if a number is undulating
    # An undulating number is one where the difference between consecutive digits is either 1 or 0
    # For simplicity, this function assumes the input is a single-digit number, as undulating typically applies to sequences of digits
    if len(str(num)) == 1:
        return abs(str(num) - str(num - 1)) in [0, 1]
    else:
        # Check each digit in the number
        for i in range(len(str(num))):
            if abs(int(str(num))[i] - int(str(num)[i + 1])) not in [0, 1]:
                return False
        return True

def is_undulating(num):
    # Function to check if a number is undulating
    # An undulating number is one where the differences between consecutive digits are either 1 or 0
    # For simplicity, we assume the input is a non-negative integer

    # Convert the number to a string to handle digit manipulation easily
    num_str = str(num)

    # Check the differences between consecutive digits
    for i in range(1, len(num_str)):
        if abs(int(num_str[i]) - int(num_str[i-1])) != 1:
            return False

    return True
